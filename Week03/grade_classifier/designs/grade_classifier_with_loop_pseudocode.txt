START

SET mark = -1

WHILE mark < 0 OR mark > 100 DO
    READ mark
    IF mark < 0 OR mark > 100 THEN
        PRINT "Invalid, try again"
    ENDIF
ENDWHILE

IF mark < 40 THEN
    PRINT "Fail"
ELSE IF mark < 60 THEN
    PRINT "Pass"
ELSE IF mark < 70 THEN
    PRINT "Merit"
ELSE
    PRINT "Distinction"
END IF

END
