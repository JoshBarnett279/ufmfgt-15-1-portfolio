DO
    DISPLAY menu options:

        2. Convert m/s to mph
        3. Exit

    SWITCH choice

        CASE 1:
            READ speed in mph
            IF speed < 0 THEN
                DISPLAY "Invalid speed"
            ELSE IF speed > 70 THEN
                DISPLAY "Speeding"
            ELSE
                DISPLAY "Valid speed"
            ENDIF
            DISPLAY "Returning to the menu"
            WAIT 0.5 seconds
            BREAK

            READ speed in m/s
